CREATE TABLE CLIENTES(
CI_CLI CHAR(10) PRIMARY KEY,
NOM_CLI VARCHAR(20) NOT NULL,
APE_CLI VARCHAR(20) NOT NULL,
DIR_CLI VARCHAR(30),
TEL_CLI CHAR(10)
);
CREATE TABLE CARGOS(
NOM_CAR VARCHAR(30) PRIMARY KEY,
SUEL_EMP NUMERIC NOT NULL
);

CREATE TABLE EMPLEADOS(
CI_EMP CHAR(10) PRIMARY KEY,
NOM_EMP VARCHAR(20) NOT NULL,
APE_EMP VARCHAR(20) NOT NULL,
CAR_EMP VARCHAR(30) NOT NULL,
FEC_NAC_EMP DATE NOT NULL,
DIR_EMP VARCHAR(30) NOT NULL,
TEL_EMP CHAR(10),
E_MAIL_EMP VARCHAR(50),
EST_CIV_EMP VARCHAR(15),
GENERO VARCHAR(1),
FOTO VARCHAR(200),
CONSTRAINT FK_EMP_CARG FOREIGN KEY(CAR_EMP) REFERENCES CARGOS(NOM_CAR)
);

CREATE TABLE PRODUCTOS(
COD_PROD SERIAL PRIMARY KEY,
NOM_PROD VARCHAR(15),
MARCA_PROD VARCHAR(15),
PRECIOXUNIDAD NUMERIC,
FEC_CADUCIDAD DATE,
EXISTENCIA NUMERIC,
FOTO VARCHAR(200)
);

CREATE TABLE FACTURAS(
NUM_FAC SERIAL PRIMARY KEY,
FEC_FAC DATE NOT NULL,
CI_CLI_FAC CHAR(10) NOT NULL,
CONSTRAINT FK_FAC_CLI FOREIGN KEY(CI_CLI_FAC) REFERENCES CLIENTES(CI_CLI),
CI_EMP_FAC CHAR(10) NOT NULL,
CONSTRAINT FK_FAC_EMP FOREIGN KEY(CI_EMP_FAC) REFERENCES EMPLEADOS(CI_EMP),
SUB_TOT_FAC NUMERIC NOT NULL,
IVA_FAC NUMERIC NOT NULL,
TOTAL_PAGAR NUMERIC NOT NULL
);

CREATE TABLE DET_FACTURAS(
NUM_FAC_P INT NOT NULL,
CONSTRAINT FK_DET_FAC FOREIGN KEY(NUM_FAC_P) REFERENCES FACTURAS(NUM_FAC),
COD_PROD_P INT NOT NULL,
CONSTRAINT FK_DET_PROD FOREIGN KEY(COD_PROD_P) REFERENCES PRODUCTOS(COD_PROD),
CAN_PROD NUMERIC NOT NULL
);

CREATE TABLE CAJA(
COD_CAJ SERIAL PRIMARY KEY,
NUM_FAC_CAJ INT NOT NULL,
SALDO_CAJ NUMERIC,
CONSTRAINT FK_ FOREIGN KEY (NUM_FAC_CAJ) REFERENCES FACTURAS(NUM_FAC)
);
